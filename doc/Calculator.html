def eval(input)
  input = @results.last + input if input[0].match(/#{OPERATOR_REGEX}/) && !@results.empty?
  infix = input.scan(/#{INFIX_REGEX}/)
  postfix = to_postfix(infix)
  result = evaluate_postfix(postfix)
  @results.push(result)
  return result
end
